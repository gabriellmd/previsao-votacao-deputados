def converte_ocupacao(dataframe):
  converted = []
  for val in dataframe['ocupacao']:
    if(val == 'VEREADOR'):
      converted.append('VEREADOR')
    elif(val == 'DEPUTADO'):
      converted.append('DEPUTADO')
    elif(val == 'MÉDICO'):
      converted.append('MÉDICO')
    elif(val == 'COMERCIANTE'):
      converted.append('COMERCIANTE')
    elif(val == 'ADVOGADO'):
      converted.append('ADVOGADO')
    elif(val == 'EMPRESÁRIO'):
      converted.append('EMPRESÁRIO')
    else:
      converted.append('OUTRO')

  dataframe['ocupacao'] = converted
  return dataframe
